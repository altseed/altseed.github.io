空のウインドウを表示する

AC-Engineで空のウインドウを表示します。

﻿
  /// <summary>
  /// 空のウインドウを表示するサンプルです。
  /// </summary>
  class Empty
  {
      [System.STAThread]
      static void Main(string[] args)
      {
          // AC-Engineを初期化する。
          ace.Engine.Initialize("Empty", 640, 480, new ace.EngineOption());
  
          // AC-Engineのウインドウが閉じられていないか確認する。
          while (ace.Engine.DoEvents())
          {
              // AC-Engineを更新する。
              ace.Engine.Update();
          }
  
          // AC-Engineの終了処理をする。
          ace.Engine.Terminate();
      }
  }
